CREATE TABLE UTENTI_RUOLI (
    CODUTE INTEGER NOT NULL,
    CODRUO INTEGER NOT NULL,
    PRIMARY KEY (CODUTE, CODRUO),
    CONSTRAINT FK_UTENTI FOREIGN KEY (CODUTE) REFERENCES UTENTI (CODUTE) ON DELETE CASCADE,
    CONSTRAINT FK_RUOLI FOREIGN KEY (CODRUO) REFERENCES RUOLI (CODRUO) ON DELETE CASCADE
);
